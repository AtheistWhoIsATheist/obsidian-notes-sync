   - title, author, date, tags
   - nihiltheism-metrics (if present)
   - bibliography path
   - custom fields
3. Parse markdown body:
   - Headers become structural elements
   - Blockquotes become abstracts or emphasized content
   - Code blocks with language hint (e.g., \`\`\`mermaid) become diagrams
   - Wikilinks [[target]] become internal references
   - Citations [@bibtexKey] become bibliography references
4. Extract conceptual entities:
   - Identify philosophical terms (dictionary lookup)
   - Extract proper names (NER)
   - Identify quotations (indented or quoted)
5. Create or update nodes:
   - If ID in frontmatter matches existing, offer merge
   - If new, create with auto-generated ID
   - Link to referenced genealogy/concepts
6. Preserve formatting:
   - Store original markdown in separate field
   - Maintain edit history

private calculateRelevance(axiom: string, text: string): number {
  // Simple keyword overlap (in production, use semantic similarity)
  const axiomWords = new Set(axiom.toLowerCase().split(/\W+/).filter(w => w.length > 3));
  const textWords = new Set(text.toLowerCase().split(/\W+/).filter(w => w.length > 3));
  
  const intersection = [...axiomWords].filter(w => textWords.has(w));
  return intersection.length / axiomWords.size;
}
